64inline bool convert(const Mavlink & rmsg, mavlink_message_t & mmsg)
65{
66 if (rmsg.payload64.size() > sizeof(mmsg.payload64) / sizeof(mmsg.payload64[0])) {
67 return false;
68 }
69
70 if (!rmsg.signature.empty() && rmsg.signature.size() != sizeof(mmsg.signature)) {
71 return false;
72 }
73
74 // [[[cog:
75 // for f in FIELD_NAMES:
76 // cog.outl(f"mmsg.{f} = rmsg.{f};")
77 // ]]]
78 mmsg.magic = rmsg.magic;
79 mmsg.len = rmsg.len;
80 mmsg.incompat_flags = rmsg.incompat_flags;
81 mmsg.compat_flags = rmsg.compat_flags;
82 mmsg.seq = rmsg.seq;
83 mmsg.sysid = rmsg.sysid;
84 mmsg.compid = rmsg.compid;
85 mmsg.msgid = rmsg.msgid;
86 mmsg.checksum = rmsg.checksum;
87 // [[[end]]] (checksum: 0b66f0fc1cd46db0f18a2429c56a6b8c)
88 std::copy(rmsg.payload64.begin(), rmsg.payload64.end(), mmsg.payload64);
89 std::copy(rmsg.signature.begin(), rmsg.signature.end(), mmsg.signature);
90
91 return true;
92}

102inline bool convert(
103 const mavlink_message_t & mmsg, Mavlink & rmsg,
104 uint8_t framing_status = Mavlink::FRAMING_OK)
105{
106 const size_t payload64_len = (mmsg.len + 7) / 8;
107
108 rmsg.framing_status = framing_status;
109
110 // [[[cog:
111 // for f in FIELD_NAMES:
112 // cog.outl(f"rmsg.{f} = mmsg.{f};")
113 // ]]]
114 rmsg.magic = mmsg.magic;
115 rmsg.len = mmsg.len;
116 rmsg.incompat_flags = mmsg.incompat_flags;
117 rmsg.compat_flags = mmsg.compat_flags;
118 rmsg.seq = mmsg.seq;
119 rmsg.sysid = mmsg.sysid;
120 rmsg.compid = mmsg.compid;
121 rmsg.msgid = mmsg.msgid;
122 rmsg.checksum = mmsg.checksum;
123 // [[[end]]] (checksum: 64ef6c1af60c622ed427e005d8ca4f2a)
124 rmsg.payload64.assign(
125 mmsg.payload64,
126 mmsg.payload64 + payload64_len);
127
128 // copy signature block only if message is signed
129 if (mmsg.incompat_flags & MAVLINK_IFLAG_SIGNED) {
130 rmsg.signature.assign(
131 mmsg.signature,
132 mmsg.signature + sizeof(mmsg.signature));
133 } else {
134 rmsg.signature.clear();
135 }
136
137 return true;
138}
