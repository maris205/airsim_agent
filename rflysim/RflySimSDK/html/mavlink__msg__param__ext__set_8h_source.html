64static inline uint16_t mavlink_msg_param_ext_set_pack(uint8_t system_id, uint8_t component_id, mavlink_message_t* msg,
65 uint8_t target_system, uint8_t target_component, const char *param_id, const char *param_value, uint8_t param_type)
66{
67#if MAVLINK_NEED_BYTE_SWAP || !MAVLINK_ALIGNED_FIELDS
68 char buf[MAVLINK_MSG_ID_PARAM_EXT_SET_LEN];
69 _mav_put_uint8_t(buf, 0, target_system);
70 _mav_put_uint8_t(buf, 1, target_component);
71 _mav_put_uint8_t(buf, 146, param_type);
72 _mav_put_char_array(buf, 2, param_id, 16);
73 _mav_put_char_array(buf, 18, param_value, 128);
74 memcpy(_MAV_PAYLOAD_NON_CONST(msg), buf, MAVLINK_MSG_ID_PARAM_EXT_SET_LEN);
75#else
76 mavlink_param_ext_set_t packet;
77 packet.target_system = target_system;
78 packet.target_component = target_component;
79 packet.param_type = param_type;
80 mav_array_memcpy(packet.param_id, param_id, sizeof(char)*16);
81 mav_array_memcpy(packet.param_value, param_value, sizeof(char)*128);
82 memcpy(_MAV_PAYLOAD_NON_CONST(msg), &packet, MAVLINK_MSG_ID_PARAM_EXT_SET_LEN);
83#endif
84
85 msg->msgid = MAVLINK_MSG_ID_PARAM_EXT_SET;
86 return mavlink_finalize_message(msg, system_id, component_id, MAVLINK_MSG_ID_PARAM_EXT_SET_MIN_LEN, MAVLINK_MSG_ID_PARAM_EXT_SET_LEN, MAVLINK_MSG_ID_PARAM_EXT_SET_CRC);
87}
